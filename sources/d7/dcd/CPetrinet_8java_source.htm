132  @Override
133  public IPetrinet connect( @NonNull final String p_id, @NonNull final String p_source, @NonNull final String p_target, @NonNull @Nonnegative final Number p_capacity )
134  {
135  final IPetrinetNode l_source = m_network.node( p_source );
136  final IPetrinetNode l_target = m_network.node( p_target );
137 
138  if ( !( l_source instanceof IPlace && l_target instanceof ITransition || l_source instanceof ITransition && l_target instanceof IPlace ) )
139  throw new RuntimeException(
140  MessageFormat.format( "source [{0}] and target [{1}] must be a place and a transition or a transition and a place", p_source, p_target )
141  );
142 
143  m_network.addedge( l_source, l_target, new CPetrinetEdge( p_id, p_capacity ) );
144  return this;
145  }

174  CSerializer( @NonNull final IGraph<IPetrinetNode, IPetrinetEdge> p_graph )
175  {
176  m_connection = p_graph.edges()
177  .map( i ->
178  {
179  final Map.Entry<IPetrinetNode, IPetrinetNode> l_endpoints = p_graph.endpoints( i );
180 
181  final Map<String, Object> l_map = new HashMap<>();
182  l_map.put( "from", l_endpoints.getKey().id() );
183  l_map.put( "to", l_endpoints.getValue().id() );
184  l_map.put( "capacity", i.get() );
185 
186  return l_map;
187  } )
188  .collect( Collectors.toSet() );
189 
190  m_places = p_graph.nodes()
191  .parallel()
192  .filter( i -> i instanceof IPlace )
193  .map( INode::<IPlace>raw )
194  .collect( Collectors.toMap( INode::id, i -> i ) );
195 
196  m_transition = p_graph.nodes()
197  .parallel()
198  .filter( i -> i instanceof ITransition )
199  .map( INode::<ITransition>raw )
200  .collect( Collectors.toMap( INode::id, i -> i ) );
201  }
202  }
